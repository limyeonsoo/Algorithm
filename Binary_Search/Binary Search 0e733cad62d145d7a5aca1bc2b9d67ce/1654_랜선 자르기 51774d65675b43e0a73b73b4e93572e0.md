```cpp
#include <iostream>
#include <algorithm>
#include <vector>
#define ll long long 

using namespace std;
ll N, K, answer;
vector <ll> lan;
void input(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cin >> N >> K; lan.resize(N);
	for(ll i=0; i<N; i++) cin >> lan[i];
}
bool check(ll value){
	ll cnt=0;
	for(ll i=0; i<lan.size(); i++){
		cnt+=lan[i]/value;
	}
	return cnt>=K;
}
void b_s(ll start, ll end){
	if(end-start < 0) return;
	ll mid = (start + end)/2;
	if(mid < 1) return;
	if(check(mid)){
		answer = answer < mid ? mid : answer;	
		b_s(mid+1, end);
	}else{
		b_s(start, mid-1);
	}
}
int main() {
	// your code goes here
	input();
	//찾으려는 것 :최대 길이
	b_s(1, *max_element(lan.begin(), lan.end()));
	cout << answer;
	return 0;
}
```

# Fail

- 자료형

    mid = (start + end) / 2 일 때,  (N + (N+1) ) / 2  가 되어 int 자료형을 넘을 수 있다.
