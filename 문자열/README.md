**< int - to - str >**

```cpp
string intToStr(int x) {
	string result = "";
	while (x > 0) {
		int end = x % 10;
		result += '0' + end;
		x /= 10;
	}
	reverse(result.begin(), result.end());
	return result;
}
```

```cpp
int strToInt(string x) {
	int result = 0;
	while (x.size()) {
		result *= 10;
		result += x[0]-'0';
		x = x.substr(1, x.size());
	}

	return result;
}
```
